Inquiries to Google Sheets in 3 Steps

1Create a Google Sheet

Open a new spreadsheet in Google Sheets. From the menu bar, click on Extensions and select Apps Script.

2Paste code & Deploy

Erase any existing code in the editor, copy the CORS-enabled code block shown on the right, and paste it. Click Deploy → New Deployment. Select type Web App, change Access to Anyone, and click Deploy.

3Save Web App URL Below

Copy the generated Web App URL from Google, paste it in the field below, and click Save. That's it! Your forms are dynamically connected.

Apps Script Code Block

This lightweight script processes submissions, appends headers automatically, and formats timestamps.

function doGet(e) {
  var action = e.parameter.action;
  if (action === 'get_articles') {
    return getArticles();
  } else if (action === 'get_gallery') {
    return getGallery();
  }
  return ContentService.createTextOutput(JSON.stringify({ "status": "error", "message": "Invalid action" }))
                       .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var action = data.action;
    
    if (action === 'add_lead') {
      return addLead(data);
    } else if (action === 'delete_lead') {
      return deleteLead(data.id);
    } else if (action === 'add_article') {
      return addArticle(data);
    } else if (action === 'delete_article') {
      return deleteArticle(data.id);
    } else if (action === 'add_gallery') {
      return addGallery(data);
    } else if (action === 'delete_gallery') {
      return deleteGallery(data.id);
    } else if (action === 'upload_to_drive') {
      return uploadToDrive(data);
    }
    
    return addLead(data);
  } catch(error) {
    return ContentService.createTextOutput(JSON.stringify({ "status": "error", "message": error.toString() }))
                         .setMimeType(ContentService.MimeType.JSON);
  }
}

function getArticles() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName("Articles");
  if (!sheet) {
    return ContentService.createTextOutput(JSON.stringify([]))
                         .setMimeType(ContentService.MimeType.JSON);
  }
  
  var data = sheet.getDataRange().getValues();
  if (data.length <= 1) {
    return ContentService.createTextOutput(JSON.stringify([]))
                         .setMimeType(ContentService.MimeType.JSON);
  }
  
  var articles = [];
  var headers = data[0];
  for (var i = 1; i < data.length; i++) {
    var row = data[i];
    var article = {};
    for (var j = 0; j < headers.length; j++) {
      var key = headers[j].toLowerCase();
      if (key === 'image url') key = 'img';
      article[key] = row[j];
    }
    articles.push(article);
  }
  
  return ContentService.createTextOutput(JSON.stringify(articles))
                       .setMimeType(ContentService.MimeType.JSON);
}

function addArticle(data) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName("Articles");
  if (!sheet) {
    sheet = ss.insertSheet("Articles");
  }
  
  if (sheet.getLastRow() === 0) {
    sheet.appendRow(["ID", "Title", "Category", "Meta", "Image URL", "Desc", "Content"]);
  }
  
  sheet.appendRow([data.id, data.title, data.category, data.meta, data.img, data.desc, data.content]);
  
  return ContentService.createTextOutput(JSON.stringify({ "status": "success" }))
                       .setMimeType(ContentService.MimeType.JSON);
}

function deleteArticle(id) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName("Articles");
  if (sheet) {
    var data = sheet.getDataRange().getValues();
    for (var i = 1; i < data.length; i++) {
      if (data[i][0] == id) {
        sheet.deleteRow(i + 1);
        break;
      }
    }
  }
  return ContentService.createTextOutput(JSON.stringify({ "status": "success" }))
                       .setMimeType(ContentService.MimeType.JSON);
}

function uploadToDrive(data) {
  try {
    var folderName = "PeoplePulse_Gallery";
    var folder;
    var folders = DriveApp.getFoldersByName(folderName);
    if (folders.hasNext()) {
      folder = folders.next();
    } else {
      folder = DriveApp.createFolder(folderName);
    }
    
    var decoded = Utilities.base64Decode(data.base64);
    var blob = Utilities.newBlob(decoded, data.mimeType, data.filename);
    var file = folder.createFile(blob);
    
    try {
      file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
    } catch(shareError) {
      Logger.log("Sharing restriction: " + shareError.toString());
    }
    
    var fileId = file.getId();
    var directUrl = "https://lh3.googleusercontent.com/d/" + fileId;
    
    return ContentService.createTextOutput(JSON.stringify({ "status": "success", "url": directUrl }))
                         .setMimeType(ContentService.MimeType.JSON);
  } catch(error) {
    return ContentService.createTextOutput(JSON.stringify({ "status": "error", "message": error.toString() }))
                         .setMimeType(ContentService.MimeType.JSON);
  }
}
